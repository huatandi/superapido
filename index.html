// V45 针对 iPhone 的强制唤醒逻辑片段 async function startScanning() { // 1. 立即激活音频上下文（iOS 必须由点击触发） if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)(); if (audioCtx.state === 'suspended') await audioCtx.resume(); const videoEl = document.getElementById('v'); document.getElementById('scanUI').style.display = 'flex'; try { // 2. 显式声明约束，增加 iOS 容错 const constraints = { video: { facingMode: { exact: "environment" } // 强制调用后置摄像头 }, audio: false }; const stream = await navigator.mediaDevices.getUserMedia(constraints); // 3. 关键：先赋值流，再显式调用 play() videoEl.srcObject = stream; // 处理 iOS 异步播放权限 let playPromise = videoEl.play(); if (playPromise !== undefined) { playPromise.then(_ => { zxing.decodeFromStream(stream, 'v', (res) => { if (res) handleBarcode(res.getText(), true); }); }).catch(error => { // 如果被拦截，尝试降级启动 console.error("iOS Play Intercepted", error); }); } } catch (err) { // 如果 exact: environment 报错（某些旧款 iPhone），尝试普通模式 const fallbackStream = await navigator.mediaDevices.getUserMedia({ video: true }); videoEl.srcObject = fallbackStream; videoEl.play(); } }
